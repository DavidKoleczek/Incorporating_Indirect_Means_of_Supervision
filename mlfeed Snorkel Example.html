Example Labeling Functions¶

1. Any tweets about job postings should be irrelevant¶

In [5]:
%%capture
# Mark any job postings as negative
@labeling_function()
def lf_jobs(x):
    job_words = ['hiring', 'job fair', 'career fair', 'hiring manager', 'openings']
    return NEGATIVE if any(s in x.lower() for s in job_words) else ABSTAIN

# Create our labeler using our one LF (just one to keep things simple)
applier = LFApplier([lf_jobs])
# Get the labels generated by lf_jobs by providing it our raw text
L_lf_jobs = applier.apply(list(data.full_text))

2. Tweets without links and with less than some number of tokens are not relevant¶

In [9]:
%%capture
minimum_tokens = 6
@labeling_function()
def lf_nolink(x):
    # Returns a tuple of a list of tokens and the number of URLs.
    x = tokenize_instance(x)
    if (x[1] == 0) and len(x[0]) <= minimum_tokens:
        return NEGATIVE
    else:
        return ABSTAIN

applier = LFApplier([lf_nolink])
L_lf_nolink = applier.apply(list(data.full_text))
